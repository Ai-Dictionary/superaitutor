const fs = require('fs');
const https = require('https');
const http = require('http');
const { URL } = require('url');

async function getTemplateSource(viewName) {
  if (viewName.startsWith('http')) {
    // Remote file: fetch via HTTP/HTTPS
    return await fetchRemoteTemplate(viewName);
  } else {
    // Local file: read from disk
    const filePath = `views/site/${viewName}.hbs`;
    return fs.readFileSync(filePath, 'utf8');
  }
}

function fetchRemoteTemplate(url) {
  return new Promise((resolve, reject) => {
    const client = url.startsWith('https') ? https : http;
    client.get(url, res => {
      let data = '';
      res.on('data', chunk => data += chunk);
      res.on('end', () => resolve(data));
    }).on('error', reject);
  });
}
