<!DOCTYPE html>
<html>
<head>
  <title>Dynamic Script Loader</title>
</head>
<body>
  <button onclick="loadScript(0)">Load A.js</button>
  <button onclick="loadScript(1)">Load B.js</button>

  <script>
    function loadScript(choice) {
      const oldScript = document.getElementById('dynamic-script');
      if (oldScript) {
        oldScript.remove();
      }

      const script = document.createElement('script');
      script.id = 'dynamic-script';
      script.type = 'text/javascript';

      // Set the correct source
      if (choice === 0) {
        script.src = '/assets/a.js';
      } else if (choice === 1) {
        script.src = '/assets/b.js';
      } else {
        console.error('Invalid choice:', choice);
        return;
      }

      // Optional: Add load/error handlers for debugging
      script.onload = () => console.log('Script loaded:', script.src);
      script.onerror = () => console.error('Failed to load script:', script.src);

      // Append to document
      document.body.appendChild(script);
    }
  </script>
</body>
</html>
